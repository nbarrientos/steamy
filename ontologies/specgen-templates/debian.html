1 Introduction

This ontology will try to cover every piece of information about packages available in Debian Archive.

1.1 Packages

Source and binary packages are modeled as versioned and unversioned elements. For each unversioned package, several versioned ones may be available.

1.2 Relationships

Both source and binary package relationships are represented by properties matching the name of the relationship.

3.1. Example

Here is a very basic document describing a source package:

<rdf:RDF>
  <rdf:Description rdf:about="http://debian.org/source/acl/2.2.47-2">
    <deb:container rdf:resource="http://debian.org/path/pool/main/a/acl"/>
    <deb:priority rdf:resource="http://idi.fundacionctic.org/steamy/debian.owl#required"/>
    <deb:uploader rdf:resource="http://debian.org/people/Anibal+Monsalve+Salazar"/>
    <foaf:page rdf:resource="http://oss.sgi.com/projects/xfs/"/>
    <rdf:type rdf:resource="http://idi.fundacionctic.org/steamy/debian.owl#Source"/>
    <deb:build-depends rdf:nodeID="OtEsEhlN24"/>
    <deb:build-depends rdf:nodeID="OtEsEhlN27"/>
    <deb:build-depends rdf:nodeID="OtEsEhlN25"/>
    <deb:build-depends rdf:nodeID="OtEsEhlN26"/>
    <deb:build-depends rdf:nodeID="OtEsEhlN28"/>
    <deb:packageName>acl</deb:packageName>
    <deb:format>1.0</deb:format>
    <deb:standardsVersion>3.7.3</deb:standardsVersion>
    <deb:versionNumber rdf:resource="http://debian.org/version/2.2.47-2"/>
    <deb:distribution rdf:resource="http://debian.org/distribution/lenny"/>
    <deb:section rdf:resource="http://debian.org/section/utils"/>
    <deb:shouldBuildIn rdf:resource="http://debian.org/arch/any"/>
    <deb:area rdf:resource="http://idi.fundacionctic.org/steamy/debian.owl#main"/>
    <deb:maintainer rdf:resource="http://debian.org/team/Debian+ACL+Team"/>
    <deb:binary rdf:resource="http://debian.org/binary/libacl1/2.2.47-2"/>
    <deb:binary rdf:resource="http://debian.org/binary/libacl1-dev/2.2.47-2"/>
    <deb:binary rdf:resource="http://debian.org/binary/acl/2.2.47-2"/>
    <rdfs:label xml:lang="en">Source: acl (2.2.47-2)</rdfs:label>
  </rdf:Description>
</rdf:RDF>
